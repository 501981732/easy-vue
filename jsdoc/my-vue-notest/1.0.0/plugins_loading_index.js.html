export default {
    install(Vue, props) {
        const LoadingPlugin = Vue.extend(LoadingComponent);

        let $vm = new LoadingPlugin({
            el: document.createElement("div")
        });

        document.body.appendChild($vm.$el);

        // $vm.show = false;
        let loading = {
            show(text) {
                $vm.value = true;

                text && ($vm.text = text);
            },
            hide() {
                $vm.value = false;
            }
        };
        // const loading = {
        //     show() {
        //         if ($vm.show) return;
        //         $vm.value = true;
        //     },
        //     hide() {
        //         $vm.value = false;
        //     }
        // };
        if (!Vue.$x) {
            Vue.$x = {
                loading
            };
        } else {
            Vue.$x.loading = loading;
        }
        // if (!Vue.$loading) {
        //     Vue.$loading = loading;
        // }
        // Vue.prototype.$loading = Vue.$loading;
        Vue.mixin({
            created() {
                this.$x = Vue.$x;
            }
        });
    }
};
